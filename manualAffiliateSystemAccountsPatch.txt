  /**
     * function dbPassEncrypt
     *
     * This function is for Encrypt strings with fixed salt key.
     *
     *
     * @author  Ruben Arias
     * @date    15/03/16
     * @param $pwd - string with the text to enctypt
     * @return $pwd encrypted
     */
    function dbPassEncrypt($pwd){

        $fixedKey=Config::get('auth.saltPwd.key');  // One fixed salt key - the same to encrypt and decrypt
        $encrypt = base64_encode(mcrypt_encrypt(MCRYPT_RIJNDAEL_256, md5($fixedKey), $pwd, MCRYPT_MODE_CBC, md5(md5($fixedKey))));
        return $encrypt; //return encrypted string

    }
    /**
     * function dbPassDecrypt
     *
     * This function is for Decrypt strings with fixed salt key.
     *
     *
     * @author  Ruben Arias
     * @date    15/03/16
     * @param $pwd - string with the text to Decrypt
     * @return $pwd Decrypted
     */
    function dbPassDecrypt($pwd){

        $fixedKey=Config::get('auth.saltPwd.key');  // One fixed salt key - the same to encrypt and decrypt
        $decrypted = rtrim(mcrypt_decrypt(MCRYPT_RIJNDAEL_256, md5($fixedKey), base64_decode($pwd), MCRYPT_MODE_CBC, md5(md5($fixedKey))), "\0");

        return $decrypted;  //return encrypted string

    }
